\begin{frame}
	\frametitle{Arithmetic Expressions: Values}

	The first step of defining a semantics is to decide on what the set of \emph{values} should be, i.e. a stopping condition.

	In \textsc{Arith}, these are the numeric constants: terms of the form \num{$n$}.
\end{frame}

\begin{frame}
	\frametitle{Arithmetic Expressions: Addition}

	\[
		\infer[(\textsc{Add-L})]
		{
			e_1 + e_2 \to e_1' + e_2
		}
		{
			e_1 \to e_1'
		}
		\qquad \quad
		\infer[(\textsc{Add-R})]
		{
			\num{n} + e_2 \to \num{n} + e_2'
		}
		{
			e_2 \to e_2'
		}
	\]
	\vspace{.5em}
	\[
		\infer[(\textsc{Add-V})]
		{
			\num{n} + \num{m} \to \num{n + m}
		}
		{}
	\]
\end{frame}

\begin{frame}
	\frametitle{Variables: Context}

	How do we keep track of the current value of a variable?
	By introducing an \emph{evaluation context}, which you can think of as a ``bag'' of data necessary to execute a program.

	In \textsc{Arith}, the context only contained the program to be executed.
	In \textsc{Arith+Var}, we must also keep track of a mapping of variable symbols to values, which we'll call the \emph{environment}:
	\[
		\ctx{ \text{program}, \text{environment} }
	\]
\end{frame}

\begin{frame}
	\frametitle{Variables: Variable Reference}

	\[
		\infer[(\textsc{Var-U})]
		{
			\ctx{ x, \sigma } \to \ctx{ \num{n}, \sigma }
		}
		{
			(x \mapsto \num{n}) \in \sigma
		}
	\]
\end{frame}

\begin{frame}
	\frametitle{Variables: Variable Definition}

	\[
		\infer[(\textsc{Var-D1})]
		{
			\ctx{ \texttt{let } x = e_1 \texttt{ in } e_2, \sigma } \to \ctx{ \texttt{let } x = e_1' \texttt{ in } e_2, \sigma }
		}
		{
			\ctx{ e_1, \sigma } \to \ctx{ e_1', \sigma }
		}
	\]
	\vspace{.5em}
	\[
		\infer[(\textsc{Var-D2})]
		{
			\ctx{ \texttt{let } x = \num{n} \texttt{ in } e_2, \sigma } \to \ctx{ \texttt{let } x = \num{n} \texttt{ in } e_2', \sigma }
		}
		{
			\ctx{ e_2, \sigma \cup \set{ x \mapsto \num{n} } } \to \ctx{ e_2', \sigma \cup \set{ x \mapsto \num{n} } }
		}
	\]
	\vspace{.5em}
	\[
		\infer[(\textsc{Var-D3})]
		{
			\ctx{ \texttt{let } x = \num{n} \texttt{ in } \num{m}, \sigma } \to \ctx{ \num{m}, \sigma }
		}
		{
		}
	\]
\end{frame}
