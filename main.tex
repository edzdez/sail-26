\begin{frame}
	\frametitle{What are Semantics?}

	There are two parts to every programming language: the \emph{syntax} and the \emph{semantics}.

	Syntax describes the \emph{structure} of code (i.e. what is an allowed sequence of symbols).
	For example, the sequence \texttt{5 + 3} is allowed, but \texttt{5 + (} is not.

	Semantics assigns \emph{meaning} to code.
	For example, we might define the meaning of \texttt{x = 5 + 3} to be ``compute the result of \texttt{5 + 3} and store it in the variable \texttt{x}.''

	Note that there may be sequences of symbols that are structurally correct but meaningless, e.g. \texttt{5 = 5 + "hello, world"}.
\end{frame}

\section{Arithmetic Expressions}

\begin{frame}
	\frametitle{Arithmetic Expressions: Syntax}

	\begin{center}
		\begin{grammar}
			\firstcase{e}{\num{n}}{Numeric constants}
			\otherform{\nonterm{e} \mathrel{\texttt{+}} \nonterm{e}}{Addition}
			\otherform{\nonterm{e} \mathrel{\texttt{*}} \nonterm{e}}{Multiplication}
		\end{grammar}
	\end{center}

	In English, an expression $e$ is:
	\vspace{-\parskip}
	\begin{itemize}
		\item a numeric constant \num{$n$} where $n$ is a number,
		\item a sum $e \mathrel{\texttt{+}} e$ where each $e$ is an expression, or
		\item a product $e \mathrel{\texttt{*}} e$ where each $e$ is an expression.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Aside: (Big-step) Operational Semantics}

	The goal of \emph{operational semantics} is to describe a program by \textbf{what it does}.

	But what does this mean?
	One idea is to think of programs as \emph{functions}:
	given some inputs, we get some output.

	Following this intuition, big-step semantics describes a program by its final result.
	For example, the program $\texttt{1 + ((2 + 3) * 4)}$ is described by the number $21$.
\end{frame}

\begin{frame}
	\frametitle{Aside: Inference Rules}

	As computer scientists, we want to describe a program's semantics by formal rules so that we can reason about them mathematically and implement them in code.

	Following the tradition of mathematical logic, semantics are described by \emph{inference rules}:
	\[
		\infer[(\textsc{Rule-Name})]
		{
			J
		}
		{
			J_1 & J_2 & \dots & J_n
		}
	\]
	In English, we say that ``$J$ holds, provided that $J_1, J_2, \dots, J_n$ all hold.''
\end{frame}

\begin{frame}
	\frametitle{Arithmetic Expressions: Semantics}

	\uncover<1,4>{\[
			\infer[(\textsc{Const})]
			{
				\num{n} \Downarrow \num{n}
			}
			{}
		\]}
	\vspace{.5em}
	\[
		\uncover<2,4>{\infer[(\textsc{Add})]
			{
				e_1 + e_2 \Downarrow \num{n + m}
			}
			{
				e_1 \Downarrow \num{n} & e_2 \Downarrow \num{m}
			}}
		\qquad
		\uncover<3,4>{\infer[(\textsc{Mul})]
			{
				e_1 * e_2 \Downarrow \num{n \cdot m}
			}
			{
				e_1 \Downarrow \num{n} & e_2 \Downarrow \num{m}
			}}
	\]
\end{frame}

\section{Adding Variables}

\begin{frame}
	\frametitle{Adding Variables: Syntax}

	\begin{center}
		\begin{grammar}
			\firstcase{e}{\num{n}}{Numeric constants}
			\otherform{\textcolor{UIUCorange}{x}}{\textcolor{UIUCorange}{Variable reference}}
			\otherform{\nonterm{e} + \nonterm{e}}{Addition}
			\otherform{\nonterm{e} * \nonterm{e}}{Multiplication}
			\otherform{\textcolor{UIUCorange}{\texttt{let $x$ = $\nonterm{e}$ in $\nonterm{e}$}}}{\textcolor{UIUCorange}{Variable definition}}
		\end{grammar}
	\end{center}

	In English, an expression $e$ is:
	\vspace{-\parskip}
	\begin{itemize}
		\item a numeric constant \num{$n$} where $n$ is a number,
		\item \textcolor{UIUCorange}{a variable reference $x$ where $x$ is an identifier,}
		\item a sum $e \mathrel{\texttt{+}} e$ where each $e$ is an expression,
		\item a product $e \mathrel{\texttt{*}} e$ where each $e$ is an expression, or
		\item \textcolor{UIUCorange}{a ``let'' expression, where we define a variable $x$.}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Adding Variables: Context}

	Now that we have variables, the output of a program depends not only on the source code but also on the \emph{environment} in which it is executed.

	For example, the output of the program \texttt{x + 5} depends on the current value of $x$!

	Hence, we introduce an \emph{execution context} to keep track of all the inputs we need.

\end{frame}

\begin{frame}
	\frametitle{Adding Variables: Context}

	Previously, the context only contained the program's source code. Now, it also contains the values assigned to each variable name:
	\[
		\ctx{ \text{program}, \text{environment} }
	\]
\end{frame}

\begin{frame}
	\frametitle{Adding Variables: Semantics}
	\[
		\uncover<2>{\infer[(\textsc{Const})]
			{
				\ctx{ \num{n}, S } \Downarrow \num{n}
			}
			{}}
		\qquad
		\infer[(\textsc{Var})]
		{
			\ctx{ x, S } \Downarrow \num{n}
		}
		{
			x = \num{n} \text{ in } S
		}
	\]
	\vspace{.5em}
	\uncover<2>{\[
			\infer[(\textsc{Add})]
			{
				\ctx{ e_1 + e_2, S } \Downarrow \num{m + n}
			}
			{
				\ctx{ e_1, S } \Downarrow \num{m} & \ctx{ e_2, S } \Downarrow \num{n}
			}
		\]
		\vspace{.5em}
		\[
			\infer[(\textsc{Mul})]
			{
				\ctx{ e_1 * e_2, S } \Downarrow \num{m \cdot n}
			}
			{
				\ctx{ e_1, S } \Downarrow \num{m} & \ctx{ e_2, S } \Downarrow \num{n}
			}
		\]}
	\vspace{.5em}
	\[
		\infer[(\textsc{Let})]
		{
            \ctx{ \texttt{let $x$ = $e_1$ in $e_2$}, S } \Downarrow \num{n}
		}
		{
            \ctx{ e_1, S } \Downarrow \num{m} & \ctx{ e_2, S \cup \set{ x = \num{m}} }\Downarrow \num{n}
		}
	\]
\end{frame}
