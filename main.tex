\begin{frame}
	\frametitle{Arithmetic Expressions: Values}

	The first step of defining a semantics is to decide on what the set of \emph{values} should be, i.e. a stopping condition.

	In \textsc{Arith}, these are the numeric constants: terms of the form \num{$n$}.
\end{frame}

\begin{frame}
	\frametitle{Arithmetic Expressions: Addition}

	\[
		\infer[(\textsc{Add-L})]
		{
			e_1 + e_2 \to e_1' + e_2
		}
		{
			e_1 \to e_1'
		}
		\qquad \quad
		\infer[(\textsc{Add-R})]
		{
			\num{n} + e_2 \to \num{n} + e_2'
		}
		{
			e_2 \to e_2'
		}
	\]
	\vspace{.5em}
	\[
		\infer[(\textsc{Add-V})]
		{
			\num{n} + \num{m} \to \num{n + m}
		}
		{}
	\]
\end{frame}

\begin{frame}
	\frametitle{Arithmetic Expressions: Multiplication}

	\[

		{
			e_1 \to e_1'
		}
		\qquad \quad
		\infer[(\textsc{Mul-R})]
		{

			e_2 \to e_2'
		}
	\]
	\vspace{.5em}
	\[
		\infer[(\textsc{Mul-V})]
		{
			\num{n} + \num{m} \to \num{n \cdot m}
		}
		{}
	\]
\end{frame}
