\noindent\textbf{Exercise 1.} Evaluate $1 + ((2+3)\times 4)$.

\vspace{1em}
\begin{center}
\scaledproof{
\infer[\blank]
  {1 + ((2+3)\times 4) \eval \blank}
  {
    \infer[\blank]{ \num{1} \eval \blank }{}
    &
    \infer[\blank]
      {(2+3)\times 4 \eval \blank}
      {
        \infer[\blank]
          {\num{2}+\num{3} \eval \blank}
          {
            \infer[\blank]{\num{2} \eval \blank}{}
            &
            \infer[\blank]{\num{3} \eval \blank}{}
          }
        &
        \infer[\blank]{\num{4} \eval \blank}{}
      }
  }
}
\end{center}

%% ---- Exercise 2 ----
\noindent\textbf{Exercise 2.} Evaluate $\mathtt{let}\ x = 4\ \mathtt{in}\ (x + 5)$ in environment $S = \{\}$.

\vspace{1em}
\begin{center}
\scaledproof{
\infer[\blank]
  {\langle \mathtt{let}\ x=4\ \mathtt{in}\ (x+5),\; \{\} \rangle \eval \blank}
  {
    \infer[\blank]
      {\langle \num{4},\; \{\} \rangle \eval \blank}
      {}
    &
    \infer[\blank]
      {\langle x+5,\; \{x \mapsto \num{4}\} \rangle \eval \blank}
      {
        \infer[\blank]
          {\langle x,\; \{x \mapsto \num{4}\} \rangle \eval \blank}
          {}
        &
        \infer[\blank]
          {\langle \num{5},\; \{x \mapsto \num{4}\} \rangle \eval \blank}
          {}
      }
  }
}
\end{center}
