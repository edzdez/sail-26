%% ---- Exercise 1 ----
\noindent\textbf{Exercise 1.} Evaluate $1 + ((2+3)\times 4)$.

\vspace{1em}
\begin{center}
	\Large
	\scaledproof{
		\infer[\rblank]
		{\J{\num{1} + ((\num{2}+\num{3})\times \num{4}) \eval \blank}}
		{
			\infer[\rblank]{\J{\num{1} \eval \blank }}{}
			&
			\infer[\rblank]
			{\J{(\num{2}+\num{3})\times \num{4} \eval \blank}}
			{
				\infer[\rblank]
				{\J{\num{2}+\num{3} \eval \blank}}
				{
					\J{\infer[\rblank]{\num{2} \eval \blank}}{}
					&
					\J{\infer[\rblank]{\num{3} \eval \blank}}{}
				}
				&
				\infer[\rblank]{\J{\num{4} \eval \blank}}{}
			}
		}
	}
\end{center}

%% ---- Exercise 2 ----
\noindent\textbf{Exercise 2.} Evaluate $\mathtt{let}\ x = 4\ \mathtt{in}\ (x + 5)$ in environment $S = \{\}$.

\vspace{1em}
\begin{center}
	\Large
	\scaledproof{
		\infer[\rblank]
		{\J{\langle \mathtt{let}\ x=\num{4}\ \mathtt{in}\ (x+\num{5}),\; \{\} \rangle \eval \blank}}
		{
			\infer[\rblank]
			{\J{\langle \num{4},\; \{\} \rangle \eval \blank}}
			{}
			&
			\infer[\rblank]
			{\J{\langle x+\num{5},\; \{x = \num{4}\} \rangle \eval \blank}}
			{
				\infer[\rblank]
				{\J{\langle x,\; \{x = \num{4}\} \rangle \eval \blank}}
				{}
				&
				\infer[\rblank]
				{\J{\left\langle \num{5},\; \{x = \num{4}\} \right\rangle \eval \blank}}
				{}
			}
		}
	}
\end{center}
